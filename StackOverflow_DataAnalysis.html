def draw_barchart(var_x,var_y,var_xticks, var_yticks, var_title, var_filename, var_color, var_opacity):
    """Follow this example to call this function:
    draw_barchart(var_title = "Title", var_x = df['labels'], var_y = df['values'], var_xticks='X axis', 
    var_yticks='Y axis', var_filename='save.png', var_color='green', var_opacity = 0.8)    """
    
    
    trace = go.Bar(
        x = var_x,
        y = var_y,
        marker=dict(
            color = (var_color),           
            reversescale = True
        ), opacity=var_opacity, textposition='auto', text=var_y, texttemplate='%{text:.2f}'
    )
    data = [trace]
    layout = go.Layout(
        title = var_title,
        xaxis=dict(
            title=var_xticks,
            tickfont=dict(
                size=11
                ,color='rgb(107, 107, 107)'
            )
        ),
        yaxis=dict(
            title=var_yticks,
            titlefont=dict(
                size=16,
                color='rgb(107, 107, 107)'
            ),
            tickfont=dict(
                size=14
                ,color='rgb(107, 107, 107)'
            )
    )
    )
    fig = go.Figure(data=data, layout=layout)
    iplot(fig, filename=var_filename)
